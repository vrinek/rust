impl<T> BTreeSet<T> where T: Ord

fn range(&'a self, min: Bound<&T>, max: Bound<&T>) -> Range<'a, T>

Constructs a double-ended iterator over a sub-range of elements in the set, starting at min, and ending at max. If min is Unbounded, then it will be treated as "negative infinity", and if max is Unbounded, then it will be treated as "positive infinity". Thus range(Unbounded, Unbounded) will yield the whole collection.

Examples

fn main() { use std::collections::BTreeSet; use std::collections::Bound::{Included, Unbounded}; let mut set = BTreeSet::new(); set.insert(3u); set.insert(5u); set.insert(8u); for &elem in set.range(Included(&4), Included(&8)) { println!("{}", elem); } assert_eq!(Some(&5u), set.range(Included(&4), Unbounded).next()); }
use std::collections::BTreeSet;
use std::collections::Bound::{Included, Unbounded};

let mut set = BTreeSet::new();
set.insert(3u);
set.insert(5u);
set.insert(8u);
for &elem in set.range(Included(&4), Included(&8)) {
    println!("{}", elem);
}
assert_eq!(Some(&5u), set.range(Included(&4), Unbounded).next());
